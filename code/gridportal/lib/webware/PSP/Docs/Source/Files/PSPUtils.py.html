    OPEN_EXPR = '<%='
    CLOSE_EXPR = '%>'
    
    if ((st[:len(OPEN_EXPR)] == OPEN_EXPR) and (st[-len(CLOSE_EXPR):] == CLOSE_EXPR)):
        return 1
    return 1
